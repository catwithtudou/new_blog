    Goе№¶еҸ‘зј–зЁӢе®һжҲҳиҜҫз¬”и®°вҖ”Etcd¶

    д»ҘдёӢдёәйёҹзӘқеӨ§дҪ¬зҡ„Go е№¶еҸ‘зј–зЁӢе®һжҲҳиҜҫ дёӯж‘ҳеҪ•зҡ„з¬”и®°

    д»Јз Ғrepo

    ETCD жҸҗдҫӣдәҶиҫғеӨҡзҡ„еҲҶеёғејҸе№¶еҸ‘еҺҹиҜӯпјҢжҜ”еҰӮеҲҶеёғејҸдә’ж–Ҙй”ҒгҖҒеҲҶеёғејҸиҜ»еҶҷй”ҒгҖҒLeaderйҖүдёҫгҖӮ

    Leader йҖүдёҫ¶

    дё»д»Һжһ¶жһ„зҡ„жңҚеҠЎиҠӮзӮ№еҲҶдёәдё»пјҲLeaderгҖҒMasterпјүе’Ңд»ҺпјҲFollowerгҖҒSlaveпјүдёӨз§Қи§’иүІгҖӮ

    дё»иҠӮзӮ№йҖҡеёёжү§иЎҢеҶҷж“ҚдҪңпјҢд»ҺиҠӮзӮ№йҖҡеёёжү§иЎҢиҜ»ж“ҚдҪңпјҢиӢҘиҜ»еҶҷйғҪеңЁдё»иҠӮзӮ№иҖҢд»ҺиҠӮзӮ№жҸҗдҫӣеӨҮд»ҪеҲҷдё»д»Һжһ¶жһ„е°ұдјҡйҖҖеҢ–жҲҗдё»еӨҮжЁЎејҸжһ¶жһ„гҖӮ

    йҖүдё»жңәеҲ¶е°ұжҳҜйҖүжӢ©дёҖдёӘиҠӮзӮ№дҪңдёәдё»иҠӮзӮ№пјҢдҝқиҜҒдё»иҠӮзӮ№зҡ„е”ҜдёҖжҖ§д»ҺиҖҢдҝқиҜҒж•°жҚ®зҡ„дёҖиҮҙжҖ§гҖӮ

    йҖүдёҫ¶

    жңүд»ҘдёӢе’ҢйҖүдё»зӣёе…ізҡ„жҸҗдҫӣзҡ„ж–№жі•пјҡ

    Go
    // жҠҠдёҖдёӘиҠӮзӮ№йҖүдёҫдёәдё»иҠӮзӮ№дё”дјҡи®ҫзҪ®дёҖдёӘеҖј
// иҝҷжҳҜдёҖдёӘйҳ»еЎһж–№жі•пјҢеңЁи°ғз”Ёе®ғзҡ„ж—¶еҖҷдјҡиў«йҳ»еЎһпјҢзӣҙеҲ°ж»Ўи¶ідёӢйқўзҡ„дёүдёӘжқЎд»¶д№ӢдёҖпјҢжүҚдјҡеҸ–ж¶Ҳйҳ»еЎһгҖӮжҲҗеҠҹеҪ“йҖүдёәдё»пјӣжӯӨж–№жі•иҝ”еӣһй”ҷиҜҜпјӣctx иў«еҸ–ж¶Ҳ
func (e *Election) Campaign(ctx context.Context,val string)error

// йҮҚж–°и®ҫзҪ® Leader зҡ„еҖјпјҢдҪҶжҳҜдёҚдјҡйҮҚж–°йҖүдё»пјҢиҝҷдёӘж–№жі•дјҡиҝ”еӣһж–°еҖји®ҫзҪ®жҲҗеҠҹжҲ–иҖ…еӨұиҙҘзҡ„дҝЎжҒҜ
func (e *Election) Proclaim(ctx context.Context, val string) error

// ејҖе§Ӣж–°дёҖж¬ЎйҖүдёҫгҖӮиҝҷдёӘж–№жі•дјҡиҝ”еӣһж–°зҡ„йҖүдёҫжҲҗеҠҹжҲ–иҖ…еӨұиҙҘзҡ„дҝЎжҒҜ
func (e *Election) Resign(ctx context.Context) (err error)

    жҹҘиҜў¶

    жңүд»ҘдёӢе’ҢжҹҘиҜўзӣёе…ізҡ„жҸҗдҫӣзҡ„ж–№жі•пјҡ

// еҸҜд»ҘдҪҝз”ЁиҝҷдёӘж–№жі•жқҘжҹҘиҜўдё»иҠӮзӮ№дҝЎжҒҜгҖӮ

// йңҖиҰҒжіЁж„Ҹзҡ„жҳҜе®ғдёҚдјҡиҝ”еӣһдё»иҠӮзӮ№зҡ„е…ЁйғЁеҺҶеҸІеҸҳеҠЁдҝЎжҒҜпјҢиҖҢжҳҜеҸӘиҝ”еӣһжңҖиҝ‘зҡ„дёҖжқЎеҸҳеҠЁдҝЎжҒҜд»ҘеҸҠд№ӢеҗҺзҡ„еҸҳеҠЁдҝЎжҒҜ
