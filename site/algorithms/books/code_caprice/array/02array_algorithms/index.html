    func removeElement(nums []int, val int) int {
    leftIdx, rightIdx := 0, len(nums)-1
    for leftIdx <= rightIdx {

        for leftIdx <= rightIdx && nums[leftIdx] != val {
            leftIdx++
        }

        for leftIdx <= rightIdx && nums[rightIdx] == val {
            rightIdx--
        }

        if leftIdx < rightIdx {
            nums[leftIdx] = nums[rightIdx]
            leftIdx++
            rightIdx--
        }
    }

    return leftIdx
}
