class MemoryManager:
    def __init__(self):
        self.history = []

    def add_interaction(self, query, response):
        self.history.append({"query": query, "response": response})

    def get_context(self, max_turns=3):
        return self.history[-max_turns:] if len(self.history) > 0 else []
